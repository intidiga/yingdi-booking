function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = sheet.getDataRange().getValues(); const booked = []; const slots = ['09:00 – 11:00','11:00 – 13:00','14:00 – 16:00','16:00 – 18:00','18:00 – 20:00']; for(let i = 1; i < data.length; i++) { if(!data[i][0]) continue; const dateVal = data[i][2]; let dateStr = ''; if(dateVal instanceof Date) { dateStr = Utilities.formatDate(dateVal, 'Asia/Taipei', 'yyyy-MM-dd'); } else { const str = String(dateVal).trim(); if(str.match(/^\d{4}-\d{2}-\d{2}$/)) { dateStr = str; } else { const d = new Date(str); if(!isNaN(d)) { dateStr = Utilities.formatDate(d, 'Asia/Taipei', 'yyyy-MM-dd'); } else { dateStr = str; } } } const slotVal = String(data[i][3]).trim(); const startTime = slotVal.substring(0,5); const endTime = slotVal.substring(8,13); slots.forEach(function(s){ const sStart = s.substring(0,5); const sEnd = s.substring(8,13); if(sStart >= startTime && sEnd <= endTime){ booked.push(dateStr+'|'+s); } }); } const result = JSON.stringify(booked); const callback = e.parameter.callback; if(callback){ return ContentService.createTextOutput(callback+'('+result+')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = JSON.parse(e.postData.contents); sheet.appendRow([data.name, data.phone, data.date, data.slot]); Logger.log('doPost triggered: ' + JSON.stringify(data)); MailApp.sendEmail({ to: 'emberstudio9898@gmail.com', subject: '【映緹影像】新預約通知', body: '你有一筆新的拍攝預約！\n\n姓名：' + data.name + '\n電話：' + data.phone + '\n日期：' + data.date + '\n時段：' + data.slot + '\n\n請盡快與客戶聯繫確認。' }); return ContentService.createTextOutput('OK'); } function testEmail() { MailApp.sendEmail({ to: 'emberstudio9898@gmail.com', subject: '測試信', body: '這是測試信，如果收到代表寄信功能正常。' }); }
